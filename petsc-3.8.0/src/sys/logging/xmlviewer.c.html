 58: PetscErrorCode PetscViewerXMLEndSection(PetscViewer viewer, const char *name)
 59: {

 63:   XMLSectionDepth -= 2;
 64:   if (XMLSectionDepth<0) XMLSectionDepth = 0;
 65:   PetscViewerASCIIPrintf(viewer, "%*s</%s>\n", XMLSectionDepth, "", name);
 66:   return(0);
 67: }
