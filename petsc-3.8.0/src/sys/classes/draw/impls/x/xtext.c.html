122: static PetscErrorCode PetscDrawXiMatchFontSize(PetscDrawXiFont *font,int w,int h)
123: {
124:   int i,max,imax,tmp;

127:   for (i=0; i<act_nfonts; i++) {
128:     if (nfonts[i].w == w && nfonts[i].h == h) {
129:       font->font_w       = w;
130:       font->font_h       = h;
