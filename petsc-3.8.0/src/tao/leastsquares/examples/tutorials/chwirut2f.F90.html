  1: !  Program usage: mpiexec -n 1 chwirut1f [-help] [all TAO options]
  2: !
  3: !  Description:  This example demonstrates use of the TAO package to solve a
  4: !  nonlinear least-squares problem on a single processor.  We minimize the
  5: !  Chwirut function:
  6: !       sum_{i=0}^{n/2-1} ( alpha*(x_{2i+1}-x_{2i}^2)^2 + (1-x_{2i})^2 )
  7: !
  8: !  The C version of this code is chwirut1.c
  9: !
 10: !/*T
 11: !  Concepts: TAO^Solving an unconstrained minimization problem
 12: !  Routines: TaoCreate();
 13: !  Routines: TaoSetType();
 14: !  Routines: TaoSetSeparableObjectiveRoutine();
 15: !  Routines: TaoSetInitialVector();
 16: !  Routines: TaoSetFromOptions();
 17: !  Routines: TaoSolve();
 18: !  Routines: TaoDestroy();
 19: !  Processors: n
 20: !T*/
 21: !
 22: ! ----------------------------------------------------------------------
 23: !
 24: #include "chwirut2f.h"

 26: ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
 27: !                   Variable declarations
 28: ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
 29: !
 30: !  See additional variable declarations in the file chwirut2f.h

 32:       PetscErrorCode   ierr    ! used to check for functions returning nonzeros
 33:       Vec              x       ! solution vector
 34:       Vec              f       ! vector of functions
 35:       Tao        tao     ! Tao context

 37: !  Note: Any user-defined Fortran routines (such as FormGradient)
 38: !  MUST be declared as external.

 40:       external FormFunction

 42: !  Initialize TAO and PETSc
 43:       call PetscInitialize(PETSC_NULL_CHARACTER,ierr)
 44:       if (ierr .ne. 0) then
 45:          print*,'Unable to initialize PETSc'
 46:          stop
 47:       endif

 49:       call MPI_Comm_size(PETSC_COMM_WORLD,size,ierr)
 50:       CHKERRA(ierr)
 51:       call MPI_Comm_rank(PETSC_COMM_WORLD,rank,ierr)
 52:       CHKERRA(ierr)

 54: !  Initialize problem parameters
 55:       call InitializeData()

 57:       if (rank .eq. 0) then
 58: !  Allocate vectors for the solution and gradient
 59:          call VecCreateSeq(PETSC_COMM_SELF,n,x,ierr)
 60:          CHKERRA(ierr)
 61:          call VecCreateSeq(PETSC_COMM_SELF,m,f,ierr)
 62:          CHKERRA(ierr)


 65: !     The TAO code begins here

 67: !     Create TAO solver
 68:          call TaoCreate(PETSC_COMM_SELF,tao,ierr)
 69:          CHKERRA(ierr)
 70:          call TaoSetType(tao,TAOPOUNDERS,ierr)
 71:          CHKERRA(ierr)

 73: !     Set routines for function, gradient, and hessian evaluation
 74:          call TaoSetSeparableObjectiveRoutine(tao,f,                    &
 75:      &        FormFunction,0,ierr)
 76:          CHKERRA(ierr)

 78: !     Optional: Set initial guess
 79:          call FormStartingPoint(x)
 80:          call TaoSetInitialVector(tao, x, ierr)
 81:          CHKERRA(ierr)


 84: !     Check for TAO command line options
 85:          call TaoSetFromOptions(tao,ierr)
 86:          CHKERRA(ierr)
 87: !     SOLVE THE APPLICATION
 88:          call TaoSolve(tao,ierr)
 89:          CHKERRA(ierr)

 91: !     Free TAO data structures
 92:          call TaoDestroy(tao,ierr)
 93:          CHKERRA(ierr)

 95: !     Free PETSc data structures
 96:          call VecDestroy(x,ierr)
 97:          CHKERRA(ierr)
 98:          call VecDestroy(f,ierr)
 99:          CHKERRA(ierr)
100:          call StopWorkers(ierr)
101:          CHKERRA(ierr)

103:       else
104:          call TaskWorker(ierr)
105:          CHKERRA(ierr)
106:       endif

108:       call PetscFinalize(ierr)
109:       end

123:       subroutine FormFunction(tao, x, f, dummy, ierr)
124: #include "chwirut2f.h"

126:       Tao        tao
127:       Vec              x,f
128:       PetscErrorCode   ierr
129:       PetscInt         dummy

131:       PetscInt         i,checkedin
132:       PetscInt         finished_tasks
133:       integer          next_task,status(MPI_STATUS_SIZE),tag,source

135: ! PETSc's VecGetArray acts differently in Fortran than it does in C.
136: ! Calling VecGetArray((Vec) X, (PetscReal) x_array(0:1), (PetscOffset) x_index, ierr)
137: ! will return an array of doubles referenced by x_array offset by x_index.
138: !  i.e.,  to reference the kth element of X, use x_array(k + x_index).
139: ! Notice that by declaring the arrays with range (0:1), we are using the C 0-indexing practice.
140:       PetscReal        f_v(0:1),x_v(0:1),fval
141:       PetscOffset      f_i,x_i

143:       0

145: !     Get pointers to vector data
146:       call VecGetArray(x,x_v,x_i,ierr)
147:       CHKERRQ(ierr)
148:       call VecGetArray(f,f_v,f_i,ierr)
149:       CHKERRQ(ierr)


152: !     Compute F(X)
153:       if (size .eq. 1) then
154:          ! Single processor
155:          do i=0,m-1
156:             call RunSimulation(x_v(x_i),i,f_v(i+f_i),ierr)
157:          enddo
158:       else
159:          ! Multiprocessor master
160:          next_task = 0
161:          finished_tasks = 0
162:          checkedin = 0

164:          do while (finished_tasks .lt. m .or. checkedin .lt. size-1)
165:             call MPI_Recv(fval,1,MPIU_SCALAR,MPI_ANY_SOURCE,               &
166:      &           MPI_ANY_TAG,PETSC_COMM_WORLD,status,ierr)
167:             tag = status(MPI_TAG)
168:             source = status(MPI_SOURCE)
169:             if (tag .eq. IDLE_TAG) then
170:                checkedin = checkedin + 1
171:             else
172:                f_v(f_i+tag) = fval
173:                finished_tasks = finished_tasks + 1
174:             endif
175:             if (next_task .lt. m) then
176:                ! Send task to worker
177:                call MPI_Send(x_v(x_i),n,MPIU_SCALAR,source,next_task,             &
178:      &              PETSC_COMM_WORLD,ierr)
179:                next_task = next_task + 1
180:             else
181:                ! Send idle message to worker
182:                call MPI_Send(x_v(x_i),n,MPIU_SCALAR,source,IDLE_TAG,              &
183:      &              PETSC_COMM_WORLD,ierr)
184:             end if
185:          enddo
186:       endif

188: !     Restore vectors
189:       call VecRestoreArray(x,x_v,x_i,ierr)
190:       CHKERRQ(ierr)
191:       call VecRestoreArray(F,f_v,f_i,ierr)
192:       CHKERRQ(ierr)
193:       return
194:       end

199:       Vec             x

201:       PetscOffset     x_i
202:       PetscErrorCode  ierr

204:       call VecGetArray(x,x_v,x_i,ierr)
205:       CHKERRQ(ierr)
206:       x_v(x_i) = 0.15
207:       x_v(x_i+1) = 0.008
208:       x_v(x_i+2) = 0.01
209:       call VecRestoreArray(x,x_v,x_i,ierr)
210:       CHKERRQ(ierr)
211:       return
212:       end

440:       subroutine TaskWorker(ierr)
441: #include "chwirut2f.h"

443:       PetscErrorCode ierr
444:       PetscReal x(n),f
445:       integer tag
446:       PetscInt index
447:       integer status(MPI_STATUS_SIZE)

449:       tag = IDLE_TAG
450:       f   = 0.0
451:       ! Send check-in message to master
452:       call MPI_Send(f,1,MPIU_SCALAR,0,IDLE_TAG,PETSC_COMM_WORLD,ierr)
453:       CHKERRQ(ierr)
454:       do while (tag .ne. DIE_TAG)
455:          call MPI_Recv(x,n,MPIU_SCALAR,0,MPI_ANY_TAG,PETSC_COMM_WORLD,     &
456:      &        status,ierr)
457:          CHKERRQ(ierr)
458:          tag = status(MPI_TAG)
459:          if (tag .eq. IDLE_TAG) then
460:             call MPI_Send(f,1,MPIU_SCALAR,0,IDLE_TAG,PETSC_COMM_WORLD,     &
461:      &           ierr)
462:             CHKERRQ(ierr)
463:          else if (tag .ne. DIE_TAG) then
464:             index = tag
465:             ! Compute local part of residual
466:             call RunSimulation(x,index,f,ierr)
467:             CHKERRQ(ierr)

469:             ! Return residual to master
470:             call MPI_Send(f,1,MPIU_SCALAR,0,tag,PETSC_COMM_WORLD,ierr)
471:             CHKERRQ(ierr)
472:          end if
473:       enddo
474:       0
475:       return
476:       end



480:       subroutine RunSimulation(x,i,f,ierr)
481: #include "chwirut2f.h"

483:       PetscReal x(n),f
484:       PetscInt i
485:       PetscErrorCode ierr
486:       f = y(i) - exp(-x(1)*t(i))/(x(2)+x(3)*t(i))
487:       0
488:       return
489:       end

491:       subroutine StopWorkers(ierr)
492: #include "chwirut2f.h"

494:       integer checkedin
495:       integer status(MPI_STATUS_SIZE)
496:       integer source
497:       PetscReal f,x(n)
498:       PetscErrorCode ierr
499:       PetscInt i

501:       checkedin=0
502:       do while (checkedin .lt. size-1)
503:          call MPI_Recv(f,1,MPIU_SCALAR,MPI_ANY_SOURCE,MPI_ANY_TAG,         &
504:      &        PETSC_COMM_WORLD,status,ierr)
505:          CHKERRQ(ierr)
506:          checkedin=checkedin+1
507:          source = status(MPI_SOURCE)
508:          do i=1,n
509:            x(i) = 0.0
510:          enddo
511:          call MPI_Send(x,n,MPIU_SCALAR,source,DIE_TAG,PETSC_COMM_WORLD,    &
512:      &        ierr)
513:          CHKERRQ(ierr)
514:       enddo
515:       ierr=0
516:       return
517:       end
