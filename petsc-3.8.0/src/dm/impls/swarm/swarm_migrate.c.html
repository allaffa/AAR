454: PETSC_EXTERN PetscErrorCode DMSwarmCollect_DMDABoundingBox(DM dm,PetscInt *globalsize)
455: {
456:   DM_Swarm *swarm = (DM_Swarm*)dm->data;
458:   DataEx de;
459:   PetscInt p,pk,npoints,*rankval,n_points_recv,n_bbox_recv,dim,neighbour_cells;
460:   PetscMPIInt rank,nrank;
461:   void *point_buffer,*recv_points;
462:   size_t sizeof_dmswarm_point,sizeof_bbox_ctx;
463:   PetscBool isdmda;
464:   CollectBBox *bbox,*recv_bbox;
465:   const PetscMPIInt *dmneighborranks;
466:   DM dmcell;

469:   MPI_Comm_rank(PetscObjectComm((PetscObject)dm),&rank);

471:   DMSwarmGetCellDM(dm,&dmcell);
472:   if (!dmcell) SETERRQ(PetscObjectComm((PetscObject)dm),PETSC_ERR_SUP,"Only valid if cell DM provided");
473:   isdmda = PETSC_FALSE;
474:   PetscObjectTypeCompare((PetscObject)dmcell,DMDA,&isdmda);
475:   if (!isdmda) SETERRQ(PetscObjectComm((PetscObject)dm),PETSC_ERR_SUP,"Only DMDA support for CollectBoundingBox");

477:   DMGetDimension(dm,&dim);
478:   sizeof_bbox_ctx = sizeof(CollectBBox);
479:   PetscMalloc1(1,&bbox);
480:   bbox->owner_rank = rank;

482:   /* compute the bounding box based on the overlapping / stenctil size */
483:   {
484:     Vec lcoor;

486:     DMGetCoordinatesLocal(dmcell,&lcoor);
487:     if (dim >= 1) {
488:       VecStrideMin(lcoor,0,NULL,&bbox->min[0]);
489:       VecStrideMax(lcoor,0,NULL,&bbox->max[0]);
490:     }
491:     if (dim >= 2) {
492:       VecStrideMin(lcoor,1,NULL,&bbox->min[1]);
493:       VecStrideMax(lcoor,1,NULL,&bbox->max[1]);
494:     }
495:     if (dim == 3) {
496:       VecStrideMin(lcoor,2,NULL,&bbox->min[2]);
497:       VecStrideMax(lcoor,2,NULL,&bbox->max[2]);
498:     }
499:   }
500:   DataBucketGetSizes(swarm->db,&npoints,NULL,NULL);
501:   *globalsize = npoints;
502:   DMSwarmGetField(dm,DMSwarmField_rank,NULL,NULL,(void**)&rankval);
503:   DataExCreate(PetscObjectComm((PetscObject)dm),0,&de);
504:   /* use DMDA neighbours */
505:   DMDAGetNeighbors(dmcell,&dmneighborranks);
506:   if (dim == 1) {
507:     neighbour_cells = 3;
508:   } else if (dim == 2) {
509:     neighbour_cells = 9;
510:   } else {
511:     neighbour_cells = 27;
512:   }
513:   DataExTopologyInitialize(de);
514:   for (p=0; p<neighbour_cells; p++) {
515:     if ( (dmneighborranks[p] >= 0) && (dmneighborranks[p] != rank) ) {
516:       DataExTopologyAddNeighbour(de,dmneighborranks[p]);
517:     }
518:   }
519:   DataExTopologyFinalize(de);
520:   DataExInitializeSendCount(de);
521:   for (p=0; p<neighbour_cells; p++) {
522:     if ( (dmneighborranks[p] >= 0) && (dmneighborranks[p] != rank) ) {
523:       DataExAddToSendCount(de,dmneighborranks[p],1);
524:     }
525:   }
526:   DataExFinalizeSendCount(de);
527:   /* send bounding boxes */
528:   DataExPackInitialize(de,sizeof_bbox_ctx);
529:   for (p=0; p<neighbour_cells; p++) {
530:     nrank = dmneighborranks[p];
531:     if ( (nrank >= 0) && (nrank != rank) ) {
532:       /* insert bbox buffer into DataExchanger */
533:       DataExPackData(de,nrank,1,bbox);
534:     }
535:   }
536:   DataExPackFinalize(de);
537:   /* recv bounding boxes */
538:   DataExBegin(de);
539:   DataExEnd(de);
540:   DataExGetRecvData(de,&n_bbox_recv,(void**)&recv_bbox);
541:   for (p=0; p<n_bbox_recv; p++) {
542:     PetscPrintf(PETSC_COMM_SELF,"[rank %d]: box from %d : range[%+1.4e,%+1.4e]x[%+1.4e,%+1.4e]\n",rank,recv_bbox[p].owner_rank,
543:            (double)recv_bbox[p].min[0],(double)recv_bbox[p].max[0],(double)recv_bbox[p].min[1],(double)recv_bbox[p].max[1]);
544:   }
545:   /* of course this is stupid as this "generic" function should have a better way to know what the coordinates are called */
546:   DataExInitializeSendCount(de);
547:   for (pk=0; pk<n_bbox_recv; pk++) {
548:     PetscReal *array_x,*array_y;

550:     DMSwarmGetField(dm,"coorx",NULL,NULL,(void**)&array_x);
551:     DMSwarmGetField(dm,"coory",NULL,NULL,(void**)&array_y);
552:     for (p=0; p<npoints; p++) {
553:       if ((array_x[p] >= recv_bbox[pk].min[0]) && (array_x[p] <= recv_bbox[pk].max[0]) ) {
554:         if ((array_y[p] >= recv_bbox[pk].min[1]) && (array_y[p] <= recv_bbox[pk].max[1]) ) {
555:           DataExAddToSendCount(de,recv_bbox[pk].owner_rank,1);
556:         }
557:       }
558:     }
559:     DMSwarmRestoreField(dm,"coory",NULL,NULL,(void**)&array_y);
560:     DMSwarmRestoreField(dm,"coorx",NULL,NULL,(void**)&array_x);
561:   }
562:   DataExFinalizeSendCount(de);
563:   DataBucketCreatePackedArray(swarm->db,&sizeof_dmswarm_point,&point_buffer);
564:   DataExPackInitialize(de,sizeof_dmswarm_point);
565:   for (pk=0; pk<n_bbox_recv; pk++) {
566:     PetscReal *array_x,*array_y;

568:     DMSwarmGetField(dm,"coorx",NULL,NULL,(void**)&array_x);
569:     DMSwarmGetField(dm,"coory",NULL,NULL,(void**)&array_y);
570:     for (p=0; p<npoints; p++) {
571:       if ((array_x[p] >= recv_bbox[pk].min[0]) && (array_x[p] <= recv_bbox[pk].max[0]) ) {
572:         if ((array_y[p] >= recv_bbox[pk].min[1]) && (array_y[p] <= recv_bbox[pk].max[1]) ) {
573:           /* copy point into buffer */
574:           DataBucketFillPackedArray(swarm->db,p,point_buffer);
575:           /* insert point buffer into DataExchanger */
576:           DataExPackData(de,recv_bbox[pk].owner_rank,1,point_buffer);
577:         }
578:       }
579:     }
580:     DMSwarmRestoreField(dm,"coory",NULL,NULL,(void**)&array_y);
581:     DMSwarmRestoreField(dm,"coorx",NULL,NULL,(void**)&array_x);
582:   }
583:   DataExPackFinalize(de);
584:   DMSwarmRestoreField(dm,DMSwarmField_rank,NULL,NULL,(void**)&rankval);
585:   DataExBegin(de);
586:   DataExEnd(de);
587:   DataExGetRecvData(de,&n_points_recv,(void**)&recv_points);
588:   DataBucketGetSizes(swarm->db,&npoints,NULL,NULL);
589:   DataBucketSetSizes(swarm->db,npoints + n_points_recv,DATA_BUCKET_BUFFER_DEFAULT);
590:   for (p=0; p<n_points_recv; p++) {
591:     void *data_p = (void*)( (char*)recv_points + p*sizeof_dmswarm_point );

593:     DataBucketInsertPackedArray(swarm->db,npoints+p,data_p);
594:   }
595:   DataBucketDestroyPackedArray(swarm->db,&point_buffer);
596:   PetscFree(bbox);
597:   DataExView(de);
598:   DataExDestroy(de);
599:   return(0);
600: }


603: /* General collection when no order, or neighbour information is provided */
604: /*
605:  User provides context and collect() method
606:  Broadcast user context

608:  for each context / rank {
609:    collect(swarm,context,n,list)
610:  }
611: */
612: PETSC_EXTERN PetscErrorCode DMSwarmCollect_General(DM dm,PetscErrorCode (*collect)(DM,void*,PetscInt*,PetscInt**),size_t ctx_size,void *ctx,PetscInt *globalsize)
613: {
614:   DM_Swarm       *swarm = (DM_Swarm*)dm->data;
616:   DataEx         de;
617:   PetscInt       p,r,npoints,n_points_recv;
618:   PetscMPIInt    commsize,rank;
619:   void           *point_buffer,*recv_points;
620:   void           *ctxlist;
621:   PetscInt       *n2collect,**collectlist;
622:   size_t         sizeof_dmswarm_point;

625:   MPI_Comm_size(PetscObjectComm((PetscObject)dm),&commsize);
626:   MPI_Comm_rank(PetscObjectComm((PetscObject)dm),&rank);
627:   DataBucketGetSizes(swarm->db,&npoints,NULL,NULL);
628:   *globalsize = npoints;
629:   /* Broadcast user context */
630:   PetscMalloc(ctx_size*commsize,&ctxlist);
631:   MPI_Allgather(ctx,ctx_size,MPI_CHAR,ctxlist,ctx_size,MPI_CHAR,PetscObjectComm((PetscObject)dm));
632:   PetscMalloc1(commsize,&n2collect);
633:   PetscMalloc1(commsize,&collectlist);
634:   for (r=0; r<commsize; r++) {
635:     PetscInt _n2collect;
636:     PetscInt *_collectlist;
637:     void     *_ctx_r;

639:     _n2collect   = 0;
640:     _collectlist = NULL;
641:     if (r != rank) { /* don't collect data from yourself */
642:       _ctx_r = (void*)( (char*)ctxlist + r * ctx_size );
643:       collect(dm,_ctx_r,&_n2collect,&_collectlist);
644:     }
645:     n2collect[r]   = _n2collect;
646:     collectlist[r] = _collectlist;
647:   }
648:   DataExCreate(PetscObjectComm((PetscObject)dm),0,&de);
649:   /* Define topology */
650:   DataExTopologyInitialize(de);
651:   for (r=0; r<commsize; r++) {
652:     if (n2collect[r] > 0) {
653:       DataExTopologyAddNeighbour(de,(PetscMPIInt)r);
654:     }
655:   }
656:   DataExTopologyFinalize(de);
657:   /* Define send counts */
658:   DataExInitializeSendCount(de);
659:   for (r=0; r<commsize; r++) {
660:     if (n2collect[r] > 0) {
661:       DataExAddToSendCount(de,r,n2collect[r]);
662:     }
663:   }
664:   DataExFinalizeSendCount(de);
665:   /* Pack data */
666:   DataBucketCreatePackedArray(swarm->db,&sizeof_dmswarm_point,&point_buffer);
667:   DataExPackInitialize(de,sizeof_dmswarm_point);
668:   for (r=0; r<commsize; r++) {
669:     for (p=0; p<n2collect[r]; p++) {
670:       DataBucketFillPackedArray(swarm->db,collectlist[r][p],point_buffer);
671:       /* insert point buffer into the data exchanger */
672:       DataExPackData(de,r,1,point_buffer);
673:     }
674:   }
675:   DataExPackFinalize(de);
676:   /* Scatter */
677:   DataExBegin(de);
678:   DataExEnd(de);
679:   /* Collect data in DMSwarm container */
680:   DataExGetRecvData(de,&n_points_recv,(void**)&recv_points);
681:   DataBucketGetSizes(swarm->db,&npoints,NULL,NULL);
682:   DataBucketSetSizes(swarm->db,npoints + n_points_recv,DATA_BUCKET_BUFFER_DEFAULT);
683:   for (p=0; p<n_points_recv; p++) {
684:     void *data_p = (void*)( (char*)recv_points + p*sizeof_dmswarm_point );

686:     DataBucketInsertPackedArray(swarm->db,npoints+p,data_p);
687:   }
688:   /* Release memory */
689:   for (r=0; r<commsize; r++) {
690:     if (collectlist[r]) PetscFree(collectlist[r]);
691:   }
692:   PetscFree(collectlist);
693:   PetscFree(n2collect);
694:   PetscFree(ctxlist);
695:   DataBucketDestroyPackedArray(swarm->db,&point_buffer);
696:   DataExView(de);
697:   DataExDestroy(de);
698:   return(0);
699: }
