464:     }
465:     MatAssemblyBegin(ScalingMat,MAT_FINAL_ASSEMBLY);
466:     MatAssemblyEnd(ScalingMat,MAT_FINAL_ASSEMBLY);
467:     MatMatMult(ScalingMat,fetidpmat_ctx->B_delta,MAT_INITIAL_MATRIX,PETSC_DEFAULT,&fetidpmat_ctx->B_Ddelta);
468:     MatDestroy(&ScalingMat);
469:   } else {
470:     MatCreate(PETSC_COMM_SELF,&fetidpmat_ctx->B_Ddelta);
471:     MatSetSizes(fetidpmat_ctx->B_Ddelta,n_local_lambda,pcis->n_B,n_local_lambda,pcis->n_B);
472:     if (!pcbddc->use_deluxe_scaling) {
473:       MatSetType(fetidpmat_ctx->B_Ddelta,MATSEQAIJ);
474:       MatSeqAIJSetPreallocation(fetidpmat_ctx->B_Ddelta,1,NULL);
475:       for (i=0;i<n_local_lambda;i++) {
476:         MatSetValue(fetidpmat_ctx->B_Ddelta,i,cols_B_delta[i],scaling_factors[i],INSERT_VALUES);
477:       }
478:       MatAssemblyBegin(fetidpmat_ctx->B_Ddelta,MAT_FINAL_ASSEMBLY);
479:       MatAssemblyEnd(fetidpmat_ctx->B_Ddelta,MAT_FINAL_ASSEMBLY);
480:     } else {
481:       /* scaling as in Klawonn-Widlund 1999 */
482:       PCBDDCDeluxeScaling deluxe_ctx = pcbddc->deluxe_ctx;
483:       PCBDDCSubSchurs     sub_schurs = pcbddc->sub_schurs;
484:       Mat                 T;
485:       PetscScalar         *W,lwork,*Bwork;
486:       const PetscInt      *idxs = NULL;
487:       PetscInt            cum,mss,*nnz;
488:       PetscBLASInt        *pivots,B_lwork,B_N,B_ierr;

490:       if (!pcbddc->deluxe_singlemat) SETERRQ(comm,PETSC_ERR_USER,"Cannot compute B_Ddelta! rerun with -pc_bddc_deluxe_singlemat");
491:       mss  = 0;
492:       PetscCalloc1(pcis->n_B,&nnz);
493:       if (sub_schurs->is_Ej_all) {
494:         ISGetIndices(sub_schurs->is_Ej_all,&idxs);
495:         for (i=0,cum=0;i<sub_schurs->n_subs;i++) {
496:           PetscInt subset_size;

498:           ISGetLocalSize(sub_schurs->is_subs[i],&subset_size);
499:           for (j=0;j<subset_size;j++) nnz[idxs[j+cum]] = subset_size;
500:           mss  = PetscMax(mss,subset_size);
501:           cum += subset_size;
502:         }
503:       }
504:       MatCreate(PETSC_COMM_SELF,&T);
505:       MatSetSizes(T,pcis->n_B,pcis->n_B,pcis->n_B,pcis->n_B);
506:       MatSetType(T,MATSEQAIJ);
507:       MatSeqAIJSetPreallocation(T,0,nnz);
508:       PetscFree(nnz);

510:       /* workspace allocation */
511:       B_lwork = 0;
512:       if (mss) {
513:         PetscScalar dummy = 1;

515:         B_lwork = -1;
516:         PetscBLASIntCast(mss,&B_N);
517:         PetscFPTrapPush(PETSC_FP_TRAP_OFF);
518:         PetscStackCallBLAS("LAPACKgetri",LAPACKgetri_(&B_N,&dummy,&B_N,&B_N,&lwork,&B_lwork,&B_ierr));
519:         PetscFPTrapPop();
520:         if (B_ierr) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_LIB,"Error in query to GETRI Lapack routine %d",(int)B_ierr);
521:         PetscBLASIntCast((PetscInt)PetscRealPart(lwork),&B_lwork);
522:       }
523:       PetscMalloc3(mss*mss,&W,mss,&pivots,B_lwork,&Bwork);

525:       for (i=0,cum=0;i<sub_schurs->n_subs;i++) {
526:         PetscScalar  *M;
527:         PetscInt     subset_size;

529:         ISGetLocalSize(sub_schurs->is_subs[i],&subset_size);
530:         PetscBLASIntCast(subset_size,&B_N);
531:         MatDenseGetArray(deluxe_ctx->seq_mat[i],&M);
532:         PetscMemcpy(W,M,subset_size*subset_size*sizeof(PetscScalar));
533:         MatDenseRestoreArray(deluxe_ctx->seq_mat[i],&M);
534:         PetscFPTrapPush(PETSC_FP_TRAP_OFF);
535:         PetscStackCallBLAS("LAPACKgetrf",LAPACKgetrf_(&B_N,&B_N,W,&B_N,pivots,&B_ierr));
536:         if (B_ierr) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_LIB,"Error in GETRF Lapack routine %d",(int)B_ierr);
537:         PetscStackCallBLAS("LAPACKgetri",LAPACKgetri_(&B_N,W,&B_N,pivots,Bwork,&B_lwork,&B_ierr));
538:         if (B_ierr) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_LIB,"Error in GETRI Lapack routine %d",(int)B_ierr);
539:         PetscFPTrapPop();
540:         /* silent static analyzer */
541:         if (!idxs) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_PLIB,"IDXS not present");
542:         MatSetValues(T,subset_size,idxs+cum,subset_size,idxs+cum,W,INSERT_VALUES);
543:         cum += subset_size;
544:       }
545:       MatAssemblyBegin(T,MAT_FINAL_ASSEMBLY);
546:       MatAssemblyEnd(T,MAT_FINAL_ASSEMBLY);
547:       MatMatTransposeMult(T,fetidpmat_ctx->B_delta,MAT_INITIAL_MATRIX,PETSC_DEFAULT,&BD1);
548:       MatMatMult(fetidpmat_ctx->B_delta,BD1,MAT_INITIAL_MATRIX,PETSC_DEFAULT,&BD2);
549:       MatDestroy(&T);
550:       PetscFree3(W,pivots,Bwork);
551:       if (sub_schurs->is_Ej_all) {
552:         ISRestoreIndices(sub_schurs->is_Ej_all,&idxs);
553:       }
554:     }
555:   }
556:   PetscFree(scaling_factors);
557:   PetscFree(cols_B_delta);

559:   /* Layout of multipliers */
560:   PetscLayoutCreate(comm,&llay);
561:   PetscLayoutSetBlockSize(llay,1);
562:   PetscLayoutSetSize(llay,fetidpmat_ctx->n_lambda);
563:   PetscLayoutSetUp(llay);
564:   PetscLayoutGetLocalSize(llay,&fetidpmat_ctx->n);

566:   /* Local work vector of multipliers */
567:   VecCreate(PETSC_COMM_SELF,&fetidpmat_ctx->lambda_local);
568:   VecSetSizes(fetidpmat_ctx->lambda_local,n_local_lambda,n_local_lambda);
569:   VecSetType(fetidpmat_ctx->lambda_local,VECSEQ);

571:   if (BD2) {
572:     ISLocalToGlobalMapping l2g;
573:     Mat                    T,TA,*pT;
574:     IS                     is;
575:     PetscInt               nl,N;
576:     BDdelta_DN             ctx;

578:     PetscLayoutGetLocalSize(llay,&nl);
579:     PetscLayoutGetSize(llay,&N);
580:     MatCreate(comm,&T);
581:     MatSetSizes(T,nl,nl,N,N);
582:     MatSetType(T,MATIS);
583:     ISLocalToGlobalMappingCreate(comm,1,n_local_lambda,l2g_indices,PETSC_COPY_VALUES,&l2g);
584:     MatSetLocalToGlobalMapping(T,l2g,l2g);
585:     ISLocalToGlobalMappingDestroy(&l2g);
586:     MatISSetLocalMat(T,BD2);
587:     MatDestroy(&BD2);
588:     MatISGetMPIXAIJ(T,MAT_INITIAL_MATRIX,&TA);
589:     MatDestroy(&T);
590:     ISCreateGeneral(comm,n_local_lambda,l2g_indices,PETSC_USE_POINTER,&is);
591:     MatCreateSubMatrices(TA,1,&is,&is,MAT_INITIAL_MATRIX,&pT);
592:     MatDestroy(&TA);
593:     ISDestroy(&is);
594:     BD2  = pT[0];
595:     PetscFree(pT);

597:     /* B_Ddelta for non-redundant multipliers with deluxe scaling */
598:     PetscNew(&ctx);
599:     MatSetType(fetidpmat_ctx->B_Ddelta,MATSHELL);
600:     MatShellSetContext(fetidpmat_ctx->B_Ddelta,(void *)ctx);
601:     MatShellSetOperation(fetidpmat_ctx->B_Ddelta,MATOP_MULT,(void (*)(void))MatMult_BDdelta_deluxe_nonred);
602:     MatShellSetOperation(fetidpmat_ctx->B_Ddelta,MATOP_MULT_TRANSPOSE,(void (*)(void))MatMultTranspose_BDdelta_deluxe_nonred);
603:     MatShellSetOperation(fetidpmat_ctx->B_Ddelta,MATOP_DESTROY,(void (*)(void))MatDestroy_BDdelta_deluxe_nonred);
604:     MatSetUp(fetidpmat_ctx->B_Ddelta);

606:     PetscObjectReference((PetscObject)BD1);
607:     ctx->BD = BD1;
608:     KSPCreate(PETSC_COMM_SELF,&ctx->kBD);
609:     KSPSetOperators(ctx->kBD,BD2,BD2);
610:     VecDuplicate(fetidpmat_ctx->lambda_local,&ctx->work);
611:     fetidpmat_ctx->deluxe_nonred = PETSC_TRUE;
612:   }
613:   MatDestroy(&BD1);
614:   MatDestroy(&BD2);

616:   /* fetidpmat sizes */
617:   fetidpmat_ctx->n += nPgl;
618:   fetidpmat_ctx->N  = fetidpmat_ctx->n_lambda+nPg;

620:   /* Global vector for FETI-DP linear system */
621:   VecCreate(comm,&fetidp_global);
622:   VecSetSizes(fetidp_global,fetidpmat_ctx->n,fetidpmat_ctx->N);
623:   VecSetType(fetidp_global,VECMPI);
624:   VecSetUp(fetidp_global);

626:   /* Decide layout for fetidp dofs: if it is a saddle point problem
627:      pressure is ordered first in the local part of the global vector
628:      of the FETI-DP linear system */
629:   if (nPg) {
630:     Vec            v;
631:     IS             IS_l2g_p,ais;
632:     PetscLayout    alay;
633:     const PetscInt *idxs,*pranges,*aranges,*lranges;
634:     PetscInt       *l2g_indices_p,rst;
635:     PetscMPIInt    rank;

637:     PetscMalloc1(nPl,&l2g_indices_p);
638:     VecGetLayout(fetidp_global,&alay);
639:     PetscLayoutGetRanges(alay,&aranges);
640:     PetscLayoutGetRanges(play,&pranges);
641:     PetscLayoutGetRanges(llay,&lranges);

643:     MPI_Comm_rank(PetscObjectComm((PetscObject)fetidp_global),&rank);
644:     ISCreateStride(PetscObjectComm((PetscObject)fetidp_global),pranges[rank+1]-pranges[rank],aranges[rank],1,&fetidpmat_ctx->pressure);
645:     ISCreateStride(PetscObjectComm((PetscObject)fetidp_global),lranges[rank+1]-lranges[rank],aranges[rank]+pranges[rank+1]-pranges[rank],1,&fetidpmat_ctx->lagrange);
646:     ISLocalToGlobalMappingGetIndices(l2gmap_p,&idxs);
647:     /* shift local to global indices for pressure */
648:     for (i=0;i<nPl;i++) {
649:       PetscInt owner;

651:       PetscLayoutFindOwner(play,idxs[i],&owner);
652:       l2g_indices_p[i] = idxs[i]-pranges[owner]+aranges[owner];
653:     }
654:     ISLocalToGlobalMappingRestoreIndices(l2gmap_p,&idxs);
655:     ISCreateGeneral(comm,nPl,l2g_indices_p,PETSC_OWN_POINTER,&IS_l2g_p);

657:     /* local to global scatter for pressure */
658:     VecScatterCreate(fetidpmat_ctx->vP,NULL,fetidp_global,IS_l2g_p,&fetidpmat_ctx->l2g_p);
659:     ISDestroy(&IS_l2g_p);

661:     /* scatter for lagrange multipliers only */
662:     VecCreate(comm,&v);
663:     VecSetType(v,VECSTANDARD);
664:     VecSetLayout(v,llay);
665:     VecSetUp(v);
666:     ISCreateGeneral(comm,n_local_lambda,l2g_indices,PETSC_COPY_VALUES,&ais);
667:     VecScatterCreate(fetidpmat_ctx->lambda_local,NULL,v,ais,&fetidpmat_ctx->l2g_lambda_only);
668:     ISDestroy(&ais);
669:     VecDestroy(&v);

671:     /* shift local to global indices for multipliers */
672:     for (i=0;i<n_local_lambda;i++) {
673:       PetscInt owner,ps;

675:       PetscLayoutFindOwner(llay,l2g_indices[i],&owner);
676:       ps = pranges[owner+1]-pranges[owner];
677:       l2g_indices[i] = l2g_indices[i]-lranges[owner]+aranges[owner]+ps;
678:     }

680:     /* scatter from alldofs to pressures global fetidp vector */
681:     PetscLayoutGetRange(alay,&rst,NULL);
682:     ISCreateStride(comm,nPgl,rst,1,&ais);
683:     VecScatterCreate(pcis->vec1_global,pP,fetidp_global,ais,&fetidpmat_ctx->g2g_p);
684:     ISDestroy(&ais);
685:   }
686:   PetscLayoutDestroy(&llay);
687:   PetscLayoutDestroy(&play);
688:   ISCreateGeneral(comm,n_local_lambda,l2g_indices,PETSC_OWN_POINTER,&IS_l2g_lambda);

690:   /* scatter from local to global multipliers */
691:   VecScatterCreate(fetidpmat_ctx->lambda_local,NULL,fetidp_global,IS_l2g_lambda,&fetidpmat_ctx->l2g_lambda);
692:   ISDestroy(&IS_l2g_lambda);
693:   ISLocalToGlobalMappingDestroy(&l2gmap_p);
694:   VecDestroy(&fetidp_global);

696:   /* Create some work vectors needed by fetidp */
697:   VecDuplicate(pcis->vec1_B,&fetidpmat_ctx->temp_solution_B);
698:   VecDuplicate(pcis->vec1_D,&fetidpmat_ctx->temp_solution_D);
699:   return(0);
700: }


703: PetscErrorCode PCBDDCSetupFETIDPPCContext(Mat fetimat, FETIDPPC_ctx fetidppc_ctx)
704: {
705:   FETIDPMat_ctx  mat_ctx;
706:   PC_BDDC        *pcbddc = (PC_BDDC*)fetidppc_ctx->pc->data;
707:   PC_IS          *pcis = (PC_IS*)fetidppc_ctx->pc->data;
708:   PetscBool      lumped = PETSC_FALSE;

712:   MatShellGetContext(fetimat,(void**)&mat_ctx);
713:   /* get references from objects created when setting up feti mat context */
714:   PetscObjectReference((PetscObject)mat_ctx->lambda_local);
715:   fetidppc_ctx->lambda_local = mat_ctx->lambda_local;
716:   PetscObjectReference((PetscObject)mat_ctx->B_Ddelta);
717:   fetidppc_ctx->B_Ddelta = mat_ctx->B_Ddelta;
718:   if (mat_ctx->deluxe_nonred) {
719:     PC               pc,mpc;
720:     BDdelta_DN       ctx;
721:     MatSolverPackage solver;
722:     const char       *prefix;

724:     MatShellGetContext(mat_ctx->B_Ddelta,&ctx);
725:     KSPSetType(ctx->kBD,KSPPREONLY);
726:     KSPGetPC(ctx->kBD,&mpc);
727:     KSPGetPC(pcbddc->ksp_D,&pc);
728:     PCSetType(mpc,PCLU);
729:     PCFactorGetMatSolverPackage(pc,(const MatSolverPackage*)&solver);
730:     if (solver) {
731:       PCFactorSetMatSolverPackage(mpc,solver);
732:     }
733:     MatGetOptionsPrefix(fetimat,&prefix);
734:     KSPSetOptionsPrefix(ctx->kBD,prefix);
735:     KSPAppendOptionsPrefix(ctx->kBD,"bddelta_");
736:     KSPSetFromOptions(ctx->kBD);
737:   }

739:   if (mat_ctx->l2g_lambda_only) {
740:     PetscObjectReference((PetscObject)mat_ctx->l2g_lambda_only);
741:     fetidppc_ctx->l2g_lambda = mat_ctx->l2g_lambda_only;
742:   } else {
743:     PetscObjectReference((PetscObject)mat_ctx->l2g_lambda);
744:     fetidppc_ctx->l2g_lambda = mat_ctx->l2g_lambda;
745:   }
746:   /* Dirichlet preconditioner */
747:   PetscOptionsGetBool(NULL,((PetscObject)fetimat)->prefix,"-pc_lumped",&lumped,NULL);
748:   if (!lumped) {
749:     IS        iV;
750:     PetscBool discrete_harmonic = PETSC_FALSE;

752:     PetscObjectQuery((PetscObject)fetidppc_ctx->pc,"__KSPFETIDP_iV",(PetscObject*)&iV);
753:     if (iV) {
754:       PetscOptionsGetBool(NULL,((PetscObject)fetimat)->prefix,"-pc_discrete_harmonic",&discrete_harmonic,NULL);
755:     }
756:     if (discrete_harmonic) {
757:       KSP             sksp;
758:       PC              pc;
759:       PCBDDCSubSchurs sub_schurs = pcbddc->sub_schurs;
760:       Mat             A_II,A_IB,A_BI;
761:       IS              iP = NULL;
762:       PetscBool       isshell,reuse = PETSC_FALSE;
763:       KSPType         ksptype;
764:       const char      *prefix;

766:       /*
767:         We constructs a Schur complement for

769:         | A_II A_ID |
770:         | A_DI A_DD |

772:         instead of

774:         | A_II  B^t_II A_ID |
775:         | B_II -C_II   B_ID |
776:         | A_DI  B^t_ID A_DD |

778:       */
779:       if (sub_schurs && sub_schurs->reuse_solver) {
780:         PetscObjectQuery((PetscObject)sub_schurs->A,"__KSPFETIDP_iP",(PetscObject*)&iP);
781:         if (iP) reuse = PETSC_TRUE;
782:       }
783:       if (!reuse) {
784:         IS       aB;
785:         PetscInt nb;
786:         ISGetLocalSize(pcis->is_B_local,&nb);
787:         ISCreateStride(PetscObjectComm((PetscObject)pcis->A_II),nb,0,1,&aB);
788:         MatCreateSubMatrix(pcis->A_II,iV,iV,MAT_INITIAL_MATRIX,&A_II);
789:         MatCreateSubMatrix(pcis->A_IB,iV,aB,MAT_INITIAL_MATRIX,&A_IB);
790:         MatCreateSubMatrix(pcis->A_BI,aB,iV,MAT_INITIAL_MATRIX,&A_BI);
791:         ISDestroy(&aB);
792:       } else {
793:         MatCreateSubMatrix(sub_schurs->A,pcis->is_I_local,pcis->is_B_local,MAT_INITIAL_MATRIX,&A_IB);
794:         MatCreateSubMatrix(sub_schurs->A,pcis->is_B_local,pcis->is_I_local,MAT_INITIAL_MATRIX,&A_BI);
795:         PetscObjectReference((PetscObject)pcis->A_II);
796:         A_II = pcis->A_II;
797:       }
798:       MatCreateSchurComplement(A_II,A_II,A_IB,A_BI,pcis->A_BB,&fetidppc_ctx->S_j);

800:       /* propagate settings of solver */
801:       MatSchurComplementGetKSP(fetidppc_ctx->S_j,&sksp);
802:       KSPGetType(pcis->ksp_D,&ksptype);
803:       KSPSetType(sksp,ksptype);
804:       KSPGetPC(pcis->ksp_D,&pc);
805:       PetscObjectTypeCompare((PetscObject)pc,PCSHELL,&isshell);
806:       if (!isshell) {
807:         MatSolverPackage solver;
808:         PCType           pctype;

810:         PCGetType(pc,&pctype);
811:         PCFactorGetMatSolverPackage(pc,(const MatSolverPackage*)&solver);
812:         KSPGetPC(sksp,&pc);
813:         PCSetType(pc,pctype);
814:         if (solver) {
815:           PCFactorSetMatSolverPackage(pc,solver);
816:         }
817:       } else {
818:         KSPGetPC(sksp,&pc);
819:         PCSetType(pc,PCLU);
820:       }
821:       MatDestroy(&A_II);
822:       MatDestroy(&A_IB);
823:       MatDestroy(&A_BI);
824:       MatGetOptionsPrefix(fetimat,&prefix);
825:       KSPSetOptionsPrefix(sksp,prefix);
826:       KSPAppendOptionsPrefix(sksp,"harmonic_");
827:       KSPSetFromOptions(sksp);
828:       if (reuse) {
829:         KSPSetPC(sksp,sub_schurs->reuse_solver->interior_solver);
830:       }
831:     } else { /* default Dirichlet preconditioner is pde-harmonic */
832:       MatCreateSchurComplement(pcis->A_II,pcis->A_II,pcis->A_IB,pcis->A_BI,pcis->A_BB,&fetidppc_ctx->S_j);
833:       MatSchurComplementSetKSP(fetidppc_ctx->S_j,pcis->ksp_D);
834:     }
835:   } else {
836:     PetscObjectReference((PetscObject)pcis->A_BB);
837:     fetidppc_ctx->S_j = pcis->A_BB;
838:   }
839:   /* saddle-point */
840:   if (mat_ctx->xPg) {
841:     PetscObjectReference((PetscObject)mat_ctx->xPg);
842:     fetidppc_ctx->xPg = mat_ctx->xPg;
843:     PetscObjectReference((PetscObject)mat_ctx->yPg);
844:     fetidppc_ctx->yPg = mat_ctx->yPg;
845:   }
846:   return(0);
847: }

849: PetscErrorCode FETIDPMatMult_Kernel(Mat fetimat, Vec x, Vec y, PetscBool trans)
850: {
851:   FETIDPMat_ctx  mat_ctx;
852:   PC_BDDC        *pcbddc;
853:   PC_IS          *pcis;

857:   MatShellGetContext(fetimat,(void**)&mat_ctx);
858:   pcis = (PC_IS*)mat_ctx->pc->data;
859:   pcbddc = (PC_BDDC*)mat_ctx->pc->data;
860:   /* Application of B_delta^T */
861:   VecSet(pcis->vec1_B,0.);
862:   VecScatterBegin(mat_ctx->l2g_lambda,x,mat_ctx->lambda_local,INSERT_VALUES,SCATTER_REVERSE);
863:   VecScatterEnd(mat_ctx->l2g_lambda,x,mat_ctx->lambda_local,INSERT_VALUES,SCATTER_REVERSE);
864:   MatMultTranspose(mat_ctx->B_delta,mat_ctx->lambda_local,pcis->vec1_B);

866:   /* Add contribution from saddle point */
867:   if (mat_ctx->l2g_p) {
868:     VecScatterBegin(mat_ctx->l2g_p,x,mat_ctx->vP,INSERT_VALUES,SCATTER_REVERSE);
869:     VecScatterEnd(mat_ctx->l2g_p,x,mat_ctx->vP,INSERT_VALUES,SCATTER_REVERSE);
870:     if (pcbddc->switch_static) {
871:       if (trans) {
872:         MatMultTranspose(mat_ctx->B_BI,mat_ctx->vP,pcis->vec1_D);
873:       } else {
874:         MatMult(mat_ctx->Bt_BI,mat_ctx->vP,pcis->vec1_D);
875:       }
876:     }
877:     if (trans) {
878:       MatMultTransposeAdd(mat_ctx->B_BB,mat_ctx->vP,pcis->vec1_B,pcis->vec1_B);
879:     } else {
880:       MatMultAdd(mat_ctx->Bt_BB,mat_ctx->vP,pcis->vec1_B,pcis->vec1_B);
881:     }
882:   } else {
883:     if (pcbddc->switch_static) {
884:       VecSet(pcis->vec1_D,0.0);
885:     }
886:   }
887:   /* Application of \widetilde{S}^-1 */
888:   PetscMemzero(pcbddc->benign_p0,pcbddc->benign_n*sizeof(PetscScalar));
889:   PCBDDCApplyInterfacePreconditioner(mat_ctx->pc,trans);
890:   PetscMemzero(pcbddc->benign_p0,pcbddc->benign_n*sizeof(PetscScalar));
891:   VecSet(y,0.0);
892:   /* Application of B_delta */
893:   MatMult(mat_ctx->B_delta,pcis->vec1_B,mat_ctx->lambda_local);
894:   /* Contribution from boundary pressures */
895:   if (mat_ctx->C) {
896:     const PetscScalar *lx;
897:     PetscScalar       *ly;

899:     /* pressures ordered first in x and y */
900:     VecGetArrayRead(x,&lx);
901:     VecGetArray(y,&ly);
902:     VecPlaceArray(mat_ctx->xPg,lx);
903:     VecPlaceArray(mat_ctx->yPg,ly);
904:     if (trans) {
905:       MatMultTranspose(mat_ctx->C,mat_ctx->xPg,mat_ctx->yPg);
906:     } else {
907:       MatMult(mat_ctx->C,mat_ctx->xPg,mat_ctx->yPg);
908:     }
909:     VecResetArray(mat_ctx->xPg);
910:     VecResetArray(mat_ctx->yPg);
911:     VecRestoreArrayRead(x,&lx);
912:     VecRestoreArray(y,&ly);
913:   }
914:   /* Add contribution from saddle point */
915:   if (mat_ctx->l2g_p) {
916:     if (trans) {
917:       MatMultTranspose(mat_ctx->Bt_BB,pcis->vec1_B,mat_ctx->vP);
918:     } else {
919:       MatMult(mat_ctx->B_BB,pcis->vec1_B,mat_ctx->vP);
920:     }
921:     if (pcbddc->switch_static) {
922:       if (trans) {
923:         MatMultTransposeAdd(mat_ctx->Bt_BI,pcis->vec1_D,mat_ctx->vP,mat_ctx->vP);
924:       } else {
925:         MatMultAdd(mat_ctx->B_BI,pcis->vec1_D,mat_ctx->vP,mat_ctx->vP);
926:       }
927:     }
928:     VecScatterBegin(mat_ctx->l2g_p,mat_ctx->vP,y,ADD_VALUES,SCATTER_FORWARD);
929:     VecScatterEnd(mat_ctx->l2g_p,mat_ctx->vP,y,ADD_VALUES,SCATTER_FORWARD);
930:   }
931:   VecScatterBegin(mat_ctx->l2g_lambda,mat_ctx->lambda_local,y,ADD_VALUES,SCATTER_FORWARD);
932:   VecScatterEnd(mat_ctx->l2g_lambda,mat_ctx->lambda_local,y,ADD_VALUES,SCATTER_FORWARD);
933:   return(0);
934: }

954: PetscErrorCode FETIDPPCApply_Kernel(PC fetipc, Vec x, Vec y, PetscBool trans)
955: {
956:   FETIDPPC_ctx   pc_ctx;
957:   PC_IS          *pcis;

961:   PCShellGetContext(fetipc,(void**)&pc_ctx);
962:   pcis = (PC_IS*)pc_ctx->pc->data;
963:   /* Application of B_Ddelta^T */
964:   VecScatterBegin(pc_ctx->l2g_lambda,x,pc_ctx->lambda_local,INSERT_VALUES,SCATTER_REVERSE);
965:   VecScatterEnd(pc_ctx->l2g_lambda,x,pc_ctx->lambda_local,INSERT_VALUES,SCATTER_REVERSE);
966:   VecSet(pcis->vec2_B,0.0);
967:   MatMultTranspose(pc_ctx->B_Ddelta,pc_ctx->lambda_local,pcis->vec2_B);
968:   /* Application of local Schur complement */
969:   if (trans) {
970:     MatMultTranspose(pc_ctx->S_j,pcis->vec2_B,pcis->vec1_B);
971:   } else {
972:     MatMult(pc_ctx->S_j,pcis->vec2_B,pcis->vec1_B);
973:   }
974:   /* Application of B_Ddelta */
975:   MatMult(pc_ctx->B_Ddelta,pcis->vec1_B,pc_ctx->lambda_local);
976:   VecSet(y,0.0);
977:   VecScatterBegin(pc_ctx->l2g_lambda,pc_ctx->lambda_local,y,ADD_VALUES,SCATTER_FORWARD);
978:   VecScatterEnd(pc_ctx->l2g_lambda,pc_ctx->lambda_local,y,ADD_VALUES,SCATTER_FORWARD);
979:   return(0);
980: }
