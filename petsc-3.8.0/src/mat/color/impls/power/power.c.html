 13:   if (mc->dist == 1) {
 14:     mp = m;
 15:   } else {
 16:     MatMatMult(m,m,MAT_INITIAL_MATRIX,2.0,&mp);

 22:   }

 29:   MatColoringSetWeightType(imc,mc->weight_type);

 32:   MatColoringDestroy(&imc);
 33:   if (mp != m) {MatDestroy(&mp);}
